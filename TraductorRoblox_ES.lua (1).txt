-- ══════════════════════════════════════════════════════════════
--              CAPA HTTP  (MULTI-EXECUTOR)
-- ══════════════════════════════════════════════════════════════
--  Detecta automáticamente qué función HTTP tiene el executor.
local function ObtenerFnHTTP()
    if request         then return request         end
    if http_request    then return http_request    end
    if syn and syn.request                then return syn.request         end
    if http and http.request              then return http.request        end
    if fluxus and fluxus.request          then return fluxus.request      end
    if getgenv().request                  then return getgenv().request   end
    return nil
end

local FnHTTP = ObtenerFnHTTP()

local function Solicitar(url)
    if not FnHTTP then return nil, "sin_http" end
    local ok, res = pcall(FnHTTP, {
        Url     = url,
        Method  = "GET",
        Headers = { ["User-Agent"] = "Mozilla/5.0 (compatible; RobloxTranslator/1.0)" },
    })
    if ok and res then
        return (res.Body or res.body), nil
    end
    return nil, "error_red"
end

-- ══════════════════════════════════════════════════════════════
--               CODIFICACIÓN DE URL
-- ══════════════════════════════════════════════════════════════
local function URLEncode(str)
    if not str then return "" end
    str = tostring(str):gsub("\n", " "):gsub("\r", "")
    str = str:gsub("[^%w%-%_%.%~%! ]", function(c)
        return ("%%%02X"):format(c:byte())
    end)
    return str:gsub(" ", "%%20")
end

-- ══════════════════════════════════════════════════════════════
--               PARSER JSON LIGERO
-- ══════════════════════════════════════════════════════════════
local function ExtraerTraduccion(json)
    if not json then return nil end
    -- MyMemory responde con "translatedText":"..."
    local t = json:match('"translatedText"%s*:%s*"(.-[^\\])"')
           or json:match('"translatedText"%s*:%s*"()"')
    if t then
        -- Decodificar escapes básicos
        t = t:gsub("\\n", "\n")
             :gsub("\\t", "\t")
             :gsub('\\"', '"')
             :gsub("\\\\", "\\")
        return t ~= "" and t or nil
    end
    return nil
end

-- ══════════════════════════════════════════════════════════════
--               NÚCLEO DE TRADUCCIÓN
-- ══════════════════════════════════════════════════════════════
--  Filtra, consulta MyMemory y guarda en caché.
local function Traducir(texto)
    if not texto or type(texto) ~= "string" then return texto end

    -- Limpiar espacios externos
    texto = texto:match("^%s*(.-)%s*$")

    -- Filtros rápidos (sin API call)
    if #texto < CONFIG.MIN_CHARS                then return texto end
    if #texto > CONFIG.MAX_CHARS                then return texto end
    if texto:match("^[%d%s%p]+$")               then return texto end  -- solo números
    if texto:match("^[%d]+[%/:][%d]+")          then return texto end  -- fechas/horas
    if texto:match("^https?://")                 then return texto end  -- URLs
    if texto:match("^[%u%d_]+$") and #texto < 6 then return texto end  -- siglas cortas

    -- Caché
    if Cache[texto] then return Cache[texto] end

    -- Llamada a la API
    local par = CONFIG.IDIOMA_ORIGEN .. "|" .. CONFIG.IDIOMA_DESTINO
    local url = "https://api.mymemory.translated.net/get?q="
                .. URLEncode(texto)
                .. "&langpair=" .. par

    local cuerpo, err = Solicitar(url)
    if err then return texto end

    local traducido = ExtraerTraduccion(cuerpo)
    if traducido and traducido ~= texto and #traducido > 0 then
        Cache[texto]     = traducido
        Contadores.enCache = Contadores.enCache + 1
        return traducido
    end

    return texto
end

        end)

-- ══════════════════════════════════════════════════════════════
--               RECOLECTOR DE INSTANCIAS DE TEXTO
-- ══════════════════════════════════════════════════════════════
local function ObtenerTextos(padre)
    local lista = {}
    local function Buscar(obj)
        local ok, hijos = pcall(function() return obj:GetChildren() end)
        if not ok then return end
        for _, hijo in ipairs(hijos) do
            if hijo:IsA("TextLabel") or hijo:IsA("TextButton") or hijo:IsA("TextBox") then
                if hijo.Text and #hijo.Text >= CONFIG.MIN_CHARS then
                    lista[#lista + 1] = hijo
                end
            end
            if #hijo:GetChildren() > 0 then
                Buscar(hijo)
            end
        end
    end
    Buscar(padre)
    return lista
end

-- ══════════════════════════════════════════════════════════════
--               ESCANEO Y TRADUCCIÓN COMPLETA
-- ══════════════════════════════════════════════════════════════
local function EscanearYTraducir()
    if Traduciendo or not Activo then return end
    Traduciendo = true

    ActualizarEstado("⟳ Escaneando...", Color3.fromRGB(255, 200, 50))

    -- Recolectar desde PlayerGui
    local instancias = ObtenerTextos(PlayerGui)

    -- Intentar también CoreGui
    pcall(function()
        local core = ObtenerTextos(game.CoreGui)
        for _, v in ipairs(core) do instancias[#instancias + 1] = v end
    end)

    Contadores.elementos = #instancias

    for _, inst in ipairs(instancias) do
        if not inst or not inst.Parent then continue end

        local textoActual = inst.Text
        if not textoActual or #textoActual < CONFIG.MIN_CHARS then continue end

        -- Registrar texto original si aún no existe
        if not TextoOriginal[inst] then
            TextoOriginal[inst] = textoActual
        end

        local orig      = TextoOriginal[inst]
        local traducido = Traducir(orig)

        if traducido ~= orig and #traducido > 0 then
            -- Aplicar directamente al elemento del juego
            if CONFIG.MODO_DIRECTO then
                local ok = pcall(function()
                    if inst.Parent and (inst.Text == orig or inst.Text == "") then
                        inst.Text = traducido
                    end
                end)
            end

            Contadores.traducidos = Contadores.traducidos + 1

            -- Registrar en el log (sin duplicados)
            if not LogCache[orig] then
                LogCache[orig] = true
                AgregarLog(orig, traducido)
            end
        end

        task.wait(0.02) -- pequeña pausa para no bloquear el cliente
    end

    ActualizarEstado("● " .. Contadores.elementos .. " elementos", Color3.fromRGB(80, 220, 120))
    Traduciendo = false
end

-- ══════════════════════════════════════════════════════════════
--               WATCHER  (DETECCIÓN EN TIEMPO REAL)
-- ══════════════════════════════════════════════════════════════
--  Se conecta a DescendantAdded y a los cambios de propiedad
--  "Text" para traducir dialogs, tooltips y popups al instante.
local function IniciarWatcher(padre)
    pcall(function()
        padre.DescendantAdded:Connect(function(desc)
            if not Activo then return end
            if not (desc:IsA("TextLabel") or desc:IsA("TextButton") or desc:IsA("TextBox")) then return end

            task.wait(0.18) -- esperar a que el juego asigne el texto

            local texto = desc.Text
            if not texto or #texto < CONFIG.MIN_CHARS then return end

            local orig     = texto
            local traducido = Traducir(orig)

            if traducido ~= orig then
                TextoOriginal[desc] = orig

                if CONFIG.MODO_DIRECTO then
                    pcall(function() desc.Text = traducido end)
                end

                Contadores.traducidos = Contadores.traducidos + 1
                ActualizarEstado("● " .. Contadores.elementos .. " elem.")

                if not LogCache[orig] then
                    LogCache[orig] = true
                    AgregarLog(orig, traducido)
                end
            end

            -- Vigilar cambios futuros en ESTE elemento
            pcall(function()
                desc:GetPropertyChangedSignal("Text"):Connect(function()
                    if not Activo or not desc.Parent then return end
                    task.wait(0.1)

                    local nuevo = desc.Text
                    if not nuevo or #nuevo < CONFIG.MIN_CHARS then return end

                    -- Si ya es la traducción guardada, ignorar
                    if TextoOriginal[desc] and Cache[TextoOriginal[desc]] == nuevo then return end

                    -- El juego actualizó el texto: retraducir
                    TextoOriginal[desc] = nuevo
                    local nuevoTrad = Traducir(nuevo)

                    if nuevoTrad ~= nuevo then
                        pcall(function() desc.Text = nuevoTrad end)
                        if not LogCache[nuevo] then
                            LogCache[nuevo] = true
                            AgregarLog(nuevo, nuevoTrad)
                        end
                    end
                end)
            end)
        end)
    end)
end

-- ══════════════════════════════════════════════════════════════
--               FUNCIÓN PRINCIPAL
-- ══════════════════════════════════════════════════════════════
local function Iniciar()
    -- Crear interfaz
    if CONFIG.MOSTRAR_PANEL then
        CrearUI()
    end

    -- Verificar soporte HTTP
    if not FnHTTP then
        ActualizarEstado("⚠ Sin HTTP  – Usa Delta/Arceus X/KRNL", Color3.fromRGB(240, 70, 70))
        warn("═══════════════════════════════════════════════════")
        warn("  [Traductor] ⚠️  No se detectó función HTTP.")
        warn("  Executors compatibles con mobile:")
        warn("    • Delta (anteriormente Hydrogen)")
        warn("    • Arceus X")
        warn("    • Fluxus")
        warn("    • Codex")
        warn("  En PC también: Synapse X, KRNL, Script-Ware")
        warn("═══════════════════════════════════════════════════")
        return
    end

    ActualizarEstado("🔗 Watchers activos...", Color3.fromRGB(255, 200, 50))

    -- Iniciar watchers de PlayerGui y CoreGui
    IniciarWatcher(PlayerGui)
    pcall(function() IniciarWatcher(game.CoreGui) end)

    -- Primer escaneo completo (dar tiempo al juego de cargar)
    task.wait(1.2)
    EscanearYTraducir()

    -- Loop de escaneos periódicos (por si algún texto se perdió)
    task.spawn(function()
        while true do
            task.wait(CONFIG.INTERVALO_SCAN)
            EscanearYTraducir()
        end
    end)

    -- Imprimir confirmación en la consola del executor
    print("═══════════════════════════════════════════════════")
    print("  🌐 Traductor EN→ES iniciado correctamente ✅")
    print("  Modo: " .. (CONFIG.MODO_DIRECTO and "Directo (reemplaza texto)" or "Solo panel"))
    print("  API: MyMemory (gratuita)")
    print("  Intervalo de escaneo: " .. CONFIG.INTERVALO_SCAN .. "s")
    print("  El panel es ARRASTRABLE con el dedo.")
    print("  ⏸ = Pausar  |  ▼ = Minimizar")
    print("═══════════════════════════════════════════════════")
end

-- Ejecutar en un hilo separado para no bloquear el executor
task.spawn(Iniciar)
